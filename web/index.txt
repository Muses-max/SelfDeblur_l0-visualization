<body>
    <div id="startbutton">
        <button id="b1" :class="todo" v-on:Click="btnClick()"> {{todo}} </button>    
    </div>
    <script>
       //这里需要让程序的启停状态, 因此data没有写成函数
       var app=new Vue({
         el:'#startbutton',
         data:{
           start:false,
           todo:"Start"
         },
         methods:{
           btnClick:function(){
                this.todo_original = this.todo;
                this.start = !this.start;
                if(!this.start)
                   this.todo = "Start";
                else
                   this.todo = "Stop";
                axios.post('/command', {
                  state: this.todo_original
                })
                .then(function (response) {
                 console.log(response);
                })
                .catch(function (error) {
                 console.log(error);
                 alert("error");
                }); //axios  
           } //end of btnClick
         }//end of methods
       })
    </script>
